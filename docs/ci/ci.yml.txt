name: CI

# Pre-merge gate: typecheck, tests, and the motion lint. These need no DB or
# secrets, so they run fast and reliably. Vercel's Git integration runs the
# production `next build` + deploy separately on push to main.
on:
  pull_request:
  push:
    branches: [main]

jobs:
  checks:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - uses: actions/setup-node@v4
        with:
          node-version: "22"
          cache: npm
      - run: npm ci
      - name: Typecheck
        run: npx tsc --noEmit
      - name: Unit tests
        run: npm run test:run
      - name: Motion lint
        run: npm run lint:motion
